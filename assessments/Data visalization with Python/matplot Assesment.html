groupby_yr=df.groupby(by='Year').mean()

mean_=groupby_yr[['BasePay','OvertimePay','OtherPay','TotalPayBenefits']]

from mpl_toolkits import mplot3d
fig = plt.figure(figsize=(8, 10))
ax = plt.axes(projection="3d")
pivot_ = mean_.pivot_table(index='Year')




rows_, columns_= pivot_.shape

xpos = np.arange(0, rows_, 1)
ypos = np.arange(0, columns_, 1)
